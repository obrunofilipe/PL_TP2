LEX
%literals   = "+-/*=()"
%ignore     = " \t\n"
%tokens     = [ 'VAR','NUMBER' ]

[a-zA-Z_][a-zA-Z0-9_]*      return('VAR', t.value)
\d+(\.\d+)?                 return('NUMBER', float(t.value))
.                           error(f"Illegal character '{t.value[0]}', [{t.lexer.lineno}]", t.lexer.skip(1) )
